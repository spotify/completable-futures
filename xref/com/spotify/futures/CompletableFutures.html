48    /**
49     * Returns a new {@link CompletableFuture} which completes to a list of all values of its input
50     * stages, if all succeed.  The list of results is in the same order as the input stages.
51     *
52     * <p> If any of the given stages complete exceptionally, then the returned future also does so,
53     * with a {@link CompletionException} holding this exception as its cause.
54     *
55     * <p> If no stages are provided, returns a future holding an empty list.
56     *
57     * @param stages the stages to combine
58     * @param <T>    the common super-type of all of the input stages, that determines the monomorphic
59     *               type of the output future
60     * @return a future that completes to a list of the results of the supplied stages
61     * @throws NullPointerException if the stages list or any of its elements are {@code null}
62     * @since 0.1.0
63     */
64    public static <T> CompletableFuture<List<T>> allAsList(
65        List<? extends CompletionStage<? extends T>> stages) {
66      // We use traditional for-loops instead of streams here for performance reasons,
67      // see AllAsListBenchmark
68  
69      @SuppressWarnings("unchecked") // generic array creation
70      final CompletableFuture<? extends T>[] all = new CompletableFuture[stages.size()];
71      for (int i = 0; i < stages.size(); i++) {
72        all[i] = stages.get(i).toCompletableFuture();
73      }
74      return CompletableFuture.allOf(all)
75          .thenApply(ignored -> {
76            final List<T> result = new ArrayList<>(all.length);
77            for (int i = 0; i < all.length; i++) {
78              result.add(all[i].join());
79            }
80            return result;
81          });
82    }

84    /**
85     * Returns a new {@link CompletableFuture} which completes to a list of values of those input
86     * stages that succeeded. The list of results is in the same order as the input stages. For failed
87     * stages, the defaultValueMapper will be called, and the value returned from that function will
88     * be put in the resulting list.
89     *
90     * <p>If no stages are provided, returns a future holding an empty list.
91     *
92     * @param stages the stages to combine.
93     * @param defaultValueMapper a function that will be called when a future completes exceptionally
94     * to provide a default value to place in the resulting list
95     * @param <T>    the common type of all of the input stages, that determines the type of the
96     *               output future
97     * @return a future that completes to a list of the results of the supplied stages
98     * @throws NullPointerException if the stages list or any of its elements are {@code null}
99     */
100   public static <T> CompletableFuture<List<T>> successfulAsList(
101       List<? extends CompletionStage<T>> stages,
102       Function<Throwable, ? extends T> defaultValueMapper) {
103     return stages.stream()
104         .map(f -> f.exceptionally(defaultValueMapper))
105         .collect(joinList());
106   }

108   /**
109    * Returns a new {@code CompletableFuture} that is already exceptionally completed with
110    * the given exception.
111    *
112    * @param throwable the exception
113    * @param <T>       an arbitrary type for the returned future; can be anything since the future
114    *                  will be exceptionally completed and thus there will never be a value of type
115    *                  {@code T}
116    * @return a future that exceptionally completed with the supplied exception
117    * @throws NullPointerException if the supplied throwable is {@code null}
118    * @since 0.1.0
119    */
120   public static <T> CompletableFuture<T> exceptionallyCompletedFuture(Throwable throwable) {
121     final CompletableFuture<T> future = new CompletableFuture<>();
122     future.completeExceptionally(throwable);
123     return future;
124   }

126   /**
127    * Collect a stream of {@link CompletionStage}s into a single future holding a list of the
128    * joined entities.
129    *
130    * <p> Usage:
131    *
132    * <pre>{@code
133    * collection.stream()
134    *     .map(this::someAsyncFunc)
135    *     .collect(joinList())
136    *     .thenApply(this::consumeList)
137    * }</pre>
138    *
139    * <p> The generated {@link CompletableFuture} will complete to a list of all entities, in the
140    * order they were encountered in the original stream.  Similar to
141    * {@link CompletableFuture#allOf(CompletableFuture[])}, if any of the input futures complete
142    * exceptionally, then the returned CompletableFuture also does so, with a
143    * {@link CompletionException} holding this exception as its cause.
144    *
145    * @param <T> the common super-type of all of the input stages, that determines the monomorphic
146    *            type of the output future
147    * @param <S> the implementation of {@link CompletionStage} that the stream contains
148    * @return a new {@link CompletableFuture} according to the rules outlined in the method
149    * description
150    * @throws NullPointerException if any future in the stream is {@code null}
151    * @since 0.1.0
152    */
153   public static <T, S extends CompletionStage<? extends T>>
154   Collector<S, ?, CompletableFuture<List<T>>> joinList() {
155     return collectingAndThen(toList(), CompletableFutures::allAsList);
156   }

158   /**
159    * Checks that a stage is completed.
160    *
161    * @param stage the {@link CompletionStage} to check
162    * @throws IllegalStateException if the stage is not completed
163    * @since 0.1.0
164    */
165   public static void checkCompleted(CompletionStage<?> stage) {
166     if (!stage.toCompletableFuture().isDone()) {
167       throw new IllegalStateException("future was not completed");
168     }
169   }

171   /**
172    * Gets the value of a completed stage.
173    *
174    * @param stage a completed {@link CompletionStage}
175    * @param <T>   the type of the value that the stage completes into
176    * @return the value of the stage if it has one
177    * @throws IllegalStateException if the stage is not completed
178    * @since 0.1.0
179    */
180   public static <T> T getCompleted(CompletionStage<T> stage) {
181     CompletableFuture<T> future = stage.toCompletableFuture();
182     checkCompleted(future);
183     return future.join();
184   }

186   /**
187    * Returns a new stage that, when this stage completes either normally or exceptionally, is
188    * executed with this stage's result and exception as arguments to the supplied function.
189    *
190    * <p> When this stage is complete, the given function is invoked with the result (or {@code null}
191    * if none) and the exception (or {@code null} if none) of this stage as arguments, and the
192    * function's result is used to complete the returned stage.
193    *
194    * <p> This differs from
195    * {@link java.util.concurrent.CompletionStage#handle(java.util.function.BiFunction)} in that the
196    * function should return a {@link java.util.concurrent.CompletionStage} rather than the value
197    * directly.
198    *
199    * @param stage the {@link CompletionStage} to compose
200    * @param fn    the function to use to compute the value of the
201    *              returned {@link CompletionStage}
202    * @param <T>   the type of the input stage's value.
203    * @param <U>   the function's return type
204    * @return the new {@link CompletionStage}
205    * @since 0.1.0
206    */
207   public static <T, U> CompletionStage<U> handleCompose(
208       CompletionStage<T> stage,
209       BiFunction<? super T, Throwable, ? extends CompletionStage<U>> fn) {
210     return dereference(stage.handle(fn));
211   }

213   /**
214    * Returns a new stage that, when this stage completes
215    * exceptionally, is executed with this stage's exception as the
216    * argument to the supplied function.  Otherwise, if this stage
217    * completes normally, then the returned stage also completes
218    * normally with the same value.
219    *
220    * <p>This differs from
221    * {@link java.util.concurrent.CompletionStage#exceptionally(java.util.function.Function)}
222    * in that the function should return a {@link java.util.concurrent.CompletionStage} rather than
223    * the value directly.
224    *
225    * @param stage the {@link CompletionStage} to compose
226    * @param fn    the function to use to compute the value of the
227    *              returned {@link CompletionStage} if this stage completed
228    *              exceptionally
229    * @param <T>   the type of the input stage's value.
230    * @return the new {@link CompletionStage}
231    * @since 0.1.0
232    */
233   public static <T> CompletionStage<T> exceptionallyCompose(
234       CompletionStage<T> stage,
235       Function<Throwable, ? extends CompletionStage<T>> fn) {
236     return dereference(wrap(stage).exceptionally(fn));
237   }

239   /**
240    * This takes a stage of a stage of a value and returns a plain stage of a value.
241    *
242    * @param stage a {@link CompletionStage} of a {@link CompletionStage} of a value
243    * @param <T>   the type of the inner stage's value.
244    * @return the {@link CompletionStage} of the value
245    * @since 0.1.0
246    */
247   public static <T> CompletionStage<T> dereference(
248       CompletionStage<? extends CompletionStage<T>> stage) {
249     return stage.thenCompose(Function.identity());
250   }

252   private static <T> CompletionStage<CompletionStage<T>> wrap(CompletionStage<T> future) {
253     //noinspection unchecked
254     return future.thenApply(CompletableFuture::completedFuture);

277    *
278    * @param a        the first stage.
279    * @param b        the second stage.
280    * @param c        the third stage.
281    * @param function the combining function.
282    * @param <R>      the type of the combining function's return value.
283    * @param <A>      the type of the first stage's value.
284    * @param <B>      the type of the second stage's value.

286    * @return a stage that completes into the return value of the supplied function.
287    * @since 0.1.0

289   public static <R, A, B, C> CompletionStage<R> combine(
290       CompletionStage<A> a, CompletionStage<B> b, CompletionStage<C> c,
291       Function3<A, B, C, R> function) {

297         .thenApply(ignored -> function.apply(af.join(), bf.join(), cf.join()));

487   /**
488    * Polls an external resource periodically until it returns a non-empty result.
489    *
490    * <p> The polling task should return {@code Optional.empty()} until it becomes available, and
491    * then {@code Optional.of(result)}.  If the polling task throws an exception or returns null,
492    * that will cause the result future to complete exceptionally.
493    *
494    * <p> Canceling the returned future will cancel the scheduled polling task as well.
495    *
496    * <p> Note that on a ScheduledThreadPoolExecutor the polling task might remain allocated for up
497    * to {@code frequency} time after completing or being cancelled.  If you have lots of polling
498    * operations or a long polling frequency, consider setting {@code removeOnCancelPolicy} to true.
499    * See {@link java.util.concurrent.ScheduledThreadPoolExecutor#setRemoveOnCancelPolicy(boolean)}.
500    *
501    * @param pollingTask     the polling task
502    * @param frequency       the frequency to run the polling task at
503    * @param executorService the executor service to schedule the polling task on
504    * @param <T>             the type of the result of the polling task, that will be returned when
505    *                        the task succeeds.
506    * @return a future completing to the result of the polling task once that becomes available
507    */
508   public static <T> CompletableFuture<T> poll(
509       final Supplier<Optional<T>> pollingTask,
510       final Duration frequency,
511       final ScheduledExecutorService executorService) {
512     final CompletableFuture<T> result = new CompletableFuture<>();
513     final ScheduledFuture<?> scheduled = executorService.scheduleAtFixedRate(
514         () -> pollTask(pollingTask, result), 0, frequency.toMillis(), TimeUnit.MILLISECONDS);
515     result.whenComplete((r, ex) -> scheduled.cancel(true));
516     return result;
517   }
518 
519   private static <T> void pollTask(
520       final Supplier<Optional<T>> pollingTask,
521       final CompletableFuture<T> resultFuture) {
522     try {
523       pollingTask.get().ifPresent(resultFuture::complete);
524     } catch (Exception ex) {
525       resultFuture.completeExceptionally(ex);
526     }
527   }
