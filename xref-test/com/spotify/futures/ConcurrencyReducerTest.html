86    @Test
87    public void testCancellation() throws Exception {
88      final ConcurrencyReducer<String> limiter = ConcurrencyReducer.create(2, 10);
89      final CompletableFuture<String> request1 = new CompletableFuture<>();
90      final CompletableFuture<String> request2 = new CompletableFuture<>();
91  
92      final CompletableFuture<String> response1 = limiter.add(job(request1));
93      final CompletableFuture<String> response2 = limiter.add(job(request2));
94  
95      final AtomicBoolean wasInvoked = new AtomicBoolean();
96      final CompletableFuture<String> response3 =
97          limiter.add(
98              () -> {
99                wasInvoked.set(true);
100               return null;
101             });
102 
103     response3.toCompletableFuture().cancel(false);
104 
105     // 1 and 2 are in progress, 3 is cancelled
106 
107     assertFalse(response1.isDone());
108     assertFalse(response2.isDone());
109     assertTrue(response3.isDone());
110     assertEquals(2, limiter.numActive());
111     assertEquals(1, limiter.numQueued());
112 
113     request2.complete("2");
114 
115     assertFalse(response1.isDone());
116     assertTrue(response2.isDone());
117     assertTrue(response3.isDone());
118     assertEquals(1, limiter.numActive());
119     assertEquals(0, limiter.numQueued());
120 
121     request1.complete("1");
122 
123     assertTrue(response1.isDone());
124     assertTrue(response2.isDone());
125     assertTrue(response3.isDone());
126     assertEquals(0, limiter.numActive());
127     assertEquals(0, limiter.numQueued());
128 
129     assertFalse(wasInvoked.get());
130   }

169   @Test
170   public void testLongRunning() throws Exception {
171     final AtomicInteger activeCount = new AtomicInteger();
172     final AtomicInteger maxCount = new AtomicInteger();
173     final int queueSize = 11;
174     final int maxConcurrency = 10;
175     final ConcurrencyReducer<String> limiter = ConcurrencyReducer.create(maxConcurrency, queueSize);
176     List<CountingJob> jobs = new ArrayList<>();
177     List<CompletableFuture<String>> responses = new ArrayList<>();
178     for (int i = 0; i < queueSize; i++) {
179       final CountingJob job = new CountingJob(limiter::numActive, maxCount);
180       jobs.add(job);
181       responses.add(limiter.add(job));
182     }
183 
184     for (int i = 0; i < jobs.size(); i++) {
185       final CountingJob job = jobs.get(i);
186       if (i % 2 == 0) {
187         job.future.complete("success");
188       } else {
189         job.future.completeExceptionally(new IllegalStateException());
190       }
191     }
192     responses.forEach(response -> assertTrue(response.isDone()));
193     assertEquals(0, activeCount.get());
194     assertEquals(0, limiter.numActive());
195     assertEquals(0, limiter.numQueued());
196     assertEquals(maxConcurrency, limiter.remainingActiveCapacity());
197     assertEquals(maxConcurrency, maxCount.get());
198     assertEquals(queueSize, limiter.remainingQueueCapacity());
199   }

236   private Callable<CompletionStage<String>> job(final CompletionStage<String> future) {
237     return () -> future;
238   }
239 
240   private static class CountingJob implements Callable<CompletionStage<String>> {
241 
242     private final Supplier<Integer> activeCount;
243     private final AtomicInteger maxCount;
244 
245     final CompletableFuture<String> future = new CompletableFuture<>();
246 
247     public CountingJob(Supplier<Integer> activeCount, AtomicInteger maxCount) {
248       this.activeCount = activeCount;
249       this.maxCount = maxCount;
250     }
251 
252     @Override
253     public CompletionStage<String> call() throws Exception {
254       if (activeCount.get() > maxCount.get()) {
255         maxCount.set(activeCount.get());
256       }
257       return future;
258     }
259   }
